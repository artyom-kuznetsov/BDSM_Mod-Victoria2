				has_province_modifier = TUR_prov_assimilatABLE_prov_modifier

			is_core = YUG
		}
		is_colonial = no
	}
	
	mean_time_to_happen = {
		months = 180
		modifier = {
			has_country_flag = industrial_revolution_flag
			factor = 0.5
		}
		modifier = {
			year = 1870
			factor = 0.2
		}
	}
	
	option = {
		name = "tur_assimilatable_EvtOptA"

			any_owned = {

			}
		}
	}
