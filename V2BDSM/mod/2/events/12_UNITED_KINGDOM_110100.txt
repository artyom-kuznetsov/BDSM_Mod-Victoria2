country_event = {
	id = 110131
	title = "Canadian_Unifier_EvtName"
	desc = "Canadian_Unifier_EvtDesc"
	picture = "canada"
	
	trigger = {
		CAN = { exists = no }
		NOT = { has_country_modifier = canadian_unifier_modifier }
		OR = {
			NOT = { has_global_flag = canada_has_player_flag }
			ai = no
		}
		vassal_of = ENG
		war = no
	}
	
	mean_time_to_happen = {
		months = 6
		modifier = {
			NOT = { has_global_flag = singleplayer_flag }
			ai = no
			factor = 0.5
		}
	}
	
	option = {
		name = "Canadian_Unifier_EvtOptA"
		prestige = 5
		random_country = {
			limit = {
				neighbour = THIS
				primary_culture = anglo_canadian
				vassal_of = ENG
			}			
			any_owned = {
				limit = {
					# Audax Validator "." Ignore_NEXT
					has_building = province_selector
				}
				# Audax Validator "." Ignore_NEXT
				province_selector = -1
			}
			all_core = { add_core = THIS }
			annex_to = THIS
		}
		add_country_modifier = {
			name = canadian_unifier_modifier duration = 1825
		}
		set_country_flag = hidden_tooltip
		random_owned = {
			limit = {
				owner = { has_country_flag = hidden_tooltip }
			}
			# Audax Validator "." Ignore_NEXT
			owner = {
				any_country = {
					limit = {
						primary_culture = anglo_canadian
					}
					add_country_modifier = {
						name = canadian_unifier_modifier duration = 1825
					}
				}
			}
		}
		clr_country_flag = hidden_tooltip
	}
