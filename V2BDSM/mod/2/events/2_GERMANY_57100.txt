country_event = {	# LOST IMPERIAL SUPREMACY CB
	id = 57100
	
	title = "HRE_LoseSphere_EvtName"
	desc = "HRE_LoseSphere_EvtDesc"
	
	is_triggered_only = yes
	
	option = {
		name = "HRE_LoseSphere_EvtOptA"
		remove_country_modifier = has_hre_spherelings_cmodifier
		FROM = {
			add_country_modifier = {
				name = debug_cmodifier
				duration = 10
			}
		}
		random_country = {
			limit = {
				has_country_flag = hre_country_cflag
				in_sphere = THIS
				ai = yes
			}
			country_event = { id = 57102 days = 1 }
		}
	}
}
