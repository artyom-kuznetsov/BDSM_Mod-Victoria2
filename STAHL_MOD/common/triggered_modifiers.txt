# Triggered modifiers are here.
# these are checked for each countries once/month and then applied.
#
# Effects are fully scriptable here.


ai_buff_trmod = {
	trigger = {
		ai = yes
	}
	icon = 9
	tax_efficiency = 0.04
	administrative_efficiency_modifier = 0.2
	education_efficiency_modifier = 0.35
	mobilisation_size = 0.04
	supply_consumption = -0.2
}


basic_militarization_not_funded_trmod = {
	trigger = {
		exists = yes
		conscription = basic_training
		NOT = { 
			ai = yes
		}
		OR = {
			NOT = { administration_spending = 0.5 }
			NOT = { military_spending = 0.5 }
		}
	}
	core_pop_consciousness_modifier = 0.01
	land_unit_start_experience = -5
	icon = 19
}
