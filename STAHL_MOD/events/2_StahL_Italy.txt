country_event = {
	title = "cu_parma_evtname"
	desc = "cu_parma_evtdesc"
	id = 9560
	trigger = {
		is_culture_group = italian
		alliance_with = PAR
		relation = { who = PAR value = 100 }
		ai = no
		NOT = { PAR = { vassal_of = THIS } }
		PAR = { NOT = { has_country_flag = rejected_confederation } }
	}
	fire_only_once = yes
	#
	mean_time_to_happen = {
		months = 3
	}
	option = {
		name = "cu_parma_evtopta"
		random_list = {
			1 = {
				create_vassal = PAR
				relation = {
					who = PAR
					value = 100
				}
				PAR = { 
					add_country_modifier = {
						name = recently_subjugated duration = 365
					}
				}
				prestige = 3
			}
			1 = {
				inherit = PAR
				PAR = { all_core = { add_core = THIS } }
				prestige = 5
			}
			1 = {
				PAR = { all_core = { add_core = THIS } }
				PAR = { set_country_flag = rejected_confederation }
			}
		}
	}
}
