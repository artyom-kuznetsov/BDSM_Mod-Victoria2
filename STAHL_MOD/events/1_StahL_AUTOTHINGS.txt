## ALASKA AUTOCOLONY
country_event = { 
	title = "event_alaska_title"
	desc = "event_alaska_desc"
	id = 11800
	trigger = {
		invention = colonial_negotiations
		owns = 5
	}
	
	fire_only_once = yes
	option = {
		name = "event_alaska_opta"
		2630 = { secede_province = THIS }
		2599 = { secede_province = THIS }
		2631 = { secede_province = THIS }
		ai_chance = {
			factor = 1
		}
	}
}

## INHERIT NORWAY
country_event = {
	title = "colonization_evtname"
	desc = "colonization_evtdesc"
	id = 11803
	trigger = {
		tag = NOR
		ai = yes
		year = 1845
		vassal_of = SWE
		NOT = { has_global_flag = NOR_inherited_flag }
	}
	option = {
		name = "colonization_evtopta"
		SWE = { inherit = THIS }
		set_global_flag = NOR_inherited_flag
	}
}
