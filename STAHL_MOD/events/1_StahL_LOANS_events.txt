country_event = {
	id = 198450
	title = FIRSTLOAN_FIRST_CALL_EVTNAME
	desc = FIRSTLOAN_FIRST_CALL_EVTDESC
	
	trigger = {
		has_country_flag = taken_first_loan_flag
		NOT = { has_country_modifier = first_loan_ticker_modifier }
		NOT = { has_country_flag = first_loan_first_ticker_expired_flag }
	}
	
	immediate = {
		set_country_flag = pending_loan_event_flag
	}
	
	option = {
		name = FIRSTLOAN_FIRST_CALL_EVTOPTA
		clr_country_flag = pending_loan_event_flag
		add_country_modifier = { name = first_loan_2_ticker_modifier duration = 1825 }
		set_country_flag = first_loan_first_ticker_expired_flag
	}
}
