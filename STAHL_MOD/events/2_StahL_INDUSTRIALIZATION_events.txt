# GIVE AGRARIAN SOCIETY MODIFIER
country_event = { 
	title = "AGRSOC_NAME"
	desc = "AGRSOC_DESC"
	id = 140800
	picture = "countryside"
	trigger = {
		OR = {
			tag = POR
			AND = {
				tag = SPA
				has_global_flag = spanish_civilwar_over_flag
			}
		}
		NOT = { has_country_flag = agrarian_society_received_flag }
	}
	
	option = {
		name = "AGRSOC_OPTA"
		set_country_flag = agrarian_society_received_flag
		set_country_flag = agrarian_society_has_flag
		add_country_modifier = { name = agrarian_country_modifier duration = -1 }
	}
}
# INDUSTRIAL REVOLUTION FOR IBERIANS
country_event = { 
	title = "AGRSOCREM_NAME"
	desc = "AGRSOCREM_DESC"
	id = 140801
	major = yes
	picture = "iron"
	trigger = {
		has_country_flag = agrarian_society_has_flag
		literacy = 0.45
		cheap_iron = 1
		interchangeable_parts = 1
		high_n_low_pressure_steam_engines = 1
		early_railroad = 1
		is_culture_group = iberian
	}
	mean_time_to_happen = { 
		months = 60
		modifier = {
			factor = 0.1
			cheap_steel = 1
		}
		modifier = {
			factor = 0.1
			iron_railroad = 1
		}
		modifier = {
			factor = 0.1
			inorganic_chemistry = 1
		}
		modifier = {
			factor = 0.1
			steam_turbine = 1
		}
	}
	
	option = {
		name = "AGRSOCREM_OPTA"
		clr_country_flag = agrarian_society_has_flag
		remove_country_modifier = agrarian_country_modifier
		add_country_modifier = { name = industrial_revolution_modifier duration = 3000 }
		set_country_flag = industrial_revolution_flag
		set_country_flag = industrial_revolution_happened_flag
		tech_school = spaniard_tech_school
	}
}
