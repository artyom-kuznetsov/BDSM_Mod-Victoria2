# IRISH FAMINE
country_event = {
	id = 12800
	title = "Irish_Famine_Evt_Name"
	desc = "Irish_Famine_Evt_Desc"
	picture = "disease"
	
	trigger = {
		NOT = { tag = IRE }
		OR = {
			owns = 254
			owns = 263
			owns = 257
			owns = 260
		}
		NOT = { has_global_flag = irish_famine_flag }
	}
	
	mean_time_to_happen = {
		months = 180
		modifier = {
			factor = 0.25
			year = 1844
		}
		modifier = {
			factor = 0.25
			year = 1848
		}
	}
	
	option = {
		name = "Irish_Famine_Evt_OptA"
		set_global_flag = irish_famine_flag
		set_country_flag = irish_famine_country_flag
		IRE = {
			all_core = {
				add_province_modifier = {
					name = great_famine_modifier
					duration = 1460
				}
			}
		}
	}
}
